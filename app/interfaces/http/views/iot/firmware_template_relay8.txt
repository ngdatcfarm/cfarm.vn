#include <WiFi.h>
#include <PubSubClient.h>
#include <ArduinoJson.h>

// ============== CONFIG ==============
#define DEVICE_CODE "YOUR_DEVICE_CODE"
const char* WIFI_SSID = "YOUR_WIFI_SSID";
const char* WIFI_PASS = "YOUR_WIFI_PASS";
const char* MQTT_SERVER = "app.cfarm.vn";
const char* MQTT_USER = "cfarm_device";
const char* MQTT_PASS = "Abc@@123";
const char* MQTT_TOPIC = "cfarm/YOUR_DEVICE_CODE";

// Relay pins (GPIO)
const int RELAY_PINS[8] = {32, 33, 25, 26, 27, 14, 12, 13};

// Interlock pairs (channel up, channel down) - điều chỉnh theo curtain setup
const int INTERLOCK_PAIRS[][2] = {{1, 2}, {3, 4}, {5, 6}, {7, 8}};
const int NUM_INTERLOCKS = 4;

// ============== GLOBALS ==============
WiFiClient espClient;
PubSubClient mqtt(espClient);
unsigned long lastHeartbeat = 0;
const unsigned long HEARTBEAT_INTERVAL = 30000; // 30 seconds
bool relayState[8] = {false};
unsigned long relayOnTime[8] = {0};
const unsigned long MAX_RELAY_ON_TIME = 600000; // 10 minutes max

// ============== SETUP ==============
void setup() {
    Serial.begin(115200);

    // Initialize relay pins
    for (int i = 0; i < 8; i++) {
        pinMode(RELAY_PINS[i], OUTPUT);
        digitalWrite(RELAY_PINS[i], HIGH); // OFF (relay is active LOW)
    }

    connectWiFi();
    connectMqtt();

    Serial.println("Ready!");
}

// ============== LOOP ==============
void loop() {
    if (!mqtt.connected()) {
        connectMqtt();
    }
    mqtt.loop();

    // Send heartbeat
    if (millis() - lastHeartbeat > HEARTBEAT_INTERVAL) {
        lastHeartbeat = millis();
        sendHeartbeat();
    }

    // Check relay timeout
    checkRelayTimeout();
}

// ============== WIFI ==============
void connectWiFi() {
    Serial.print("Connecting to WiFi");
    WiFi.begin(WIFI_SSID, WIFI_PASS);
    while (WiFi.status() != WL_CONNECTED) {
        delay(500);
        Serial.print(".");
    }
    Serial.println("");
    Serial.print("IP: ");
    Serial.println(WiFi.localIP());
}

// ============== MQTT ==============
void connectMqtt() {
    mqtt.setServer(MQTT_SERVER, 1883);
    mqtt.setCallback(mqttCallback);

    String clientId = "ESP32_" + String(DEVICE_CODE);
    String willTopic = String(MQTT_TOPIC) + "/status";
    String willPayload = "{\"device\":\"" + String(DEVICE_CODE) + "\",\"status\":\"offline\"}";

    while (!mqtt.connected()) {
        if (mqtt.connect(clientId.c_str(), MQTT_USER, MQTT_PASS, willTopic.c_str(), 1, true, willPayload.c_str())) {
            Serial.println("MQTT connected!");
            mqtt.subscribe(String(MQTT_TOPIC).c_str());
            sendHeartbeat();
        } else {
            Serial.print("MQTT failed, retrying...");
            delay(1000);
        }
    }
}

void mqttCallback(char* topic, byte* payload, unsigned int length) {
    String msg;
    for (int i = 0; i < length; i++) {
        msg += (char)payload[i];
    }

    Serial.print("Message arrived [");
    Serial.print(topic);
    Serial.print("]: ");
    Serial.println(msg);

    // Parse JSON
    StaticJsonDocument<256> doc;
    DeserializationError error = deserializeJson(doc, msg);

    if (error) {
        Serial.print("JSON parse failed: ");
        Serial.println(error.c_str());
        return;
    }

    String action = doc["action"] | "";

    if (action == "relay") {
        int channel = doc["channel"] | 0;
        String state = doc["state"] | "";
        int duration = doc["duration"] | 0;

        if (channel >= 1 && channel <= 8) {
            setRelay(channel - 1, state == "on", duration);
        }
    } else if (action == "all") {
        String allState = doc["state"] | "";
        for (int i = 0; i < 8; i++) {
            setRelay(i, allState == "on", 0);
        }
    }
}

void sendHeartbeat() {
    String payload = "{"
        "\"device\":\"" + String(DEVICE_CODE) + "\","
        "\"status\":\"online\","
        "\"wifi_rssi\":" + String(WiFi.RSSI()) + ","
        "\"ip\":\"" + WiFi.localIP().toString() + "\","
        "\"uptime\":" + String(millis() / 1000) + ","
        "\"heap\":" + String(ESP.getFreeHeap()) +
    "}";

    mqtt.publish(String(MQTT_TOPIC).c_str(), payload.c_str(), true);
}

// ============== RELAY CONTROL ==============
void setRelay(int channel, bool on, int durationSeconds) {
    if (channel < 0 || channel >= 8) return;

    // Check interlock
    if (on) {
        for (int i = 0; i < NUM_INTERLOCKS; i++) {
            int upCh = INTERLOCK_PAIRS[i][0] - 1;
            int downCh = INTERLOCK_PAIRS[i][1] - 1;

            if (channel == upCh && relayState[downCh]) {
                Serial.println("Interlock: cannot turn ON, opposite is ON");
                return;
            }
            if (channel == downCh && relayState[upCh]) {
                Serial.println("Interlock: cannot turn ON, opposite is ON");
                return;
            }
        }
    }

    digitalWrite(RELAY_PINS[channel], on ? LOW : HIGH);
    relayState[channel] = on;
    relayOnTime[channel] = on ? millis() : 0;

    Serial.print("Relay ");
    Serial.print(channel + 1);
    Serial.println(on ? " ON" : " OFF");

    // Publish state change
    String payload = "{"
        "\"device\":\"" + String(DEVICE_CODE) + "\","
        "\"channel\":" + String(channel + 1) + ","
        "\"state\":\"" + String(on ? "on" : "off") + "\""
    "}";
    mqtt.publish((String(MQTT_TOPIC) + "/state").c_str(), payload.c_str());
}

void checkRelayTimeout() {
    for (int i = 0; i < 8; i++) {
        if (relayState[i] && relayOnTime[i] > 0) {
            if (millis() - relayOnTime[i] > MAX_RELAY_ON_TIME) {
                Serial.print("Relay ");
                Serial.print(i + 1);
                Serial.println(" timeout - turning OFF");
                setRelay(i, false, 0);
            }
        }
    }
}
